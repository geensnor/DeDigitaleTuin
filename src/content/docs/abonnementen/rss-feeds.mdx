import feeds from "../../data/feeds.json";

Hieronder staan de RSS feeds die we lezen, per categorie. Het overzicht wordt gegeneereerd vanuit [Feedly](https://feedly.com).

<ul>
  {feeds.opml.body.outline.map((group) => (
    <li key={group.title}>
      {group.title}
      {Array.isArray(group.outline) ? (
        <ul>
          {group.outline.map((source) => (
            <li key={source.title}>
              <a target="_blank" href={source.htmlUrl}>{source.title}</a>
            </li>
          ))}
        </ul>
      ) : group.outline.title ? (
        <ul>
          <li>
            <a target="_blank" href={group.outline.htmlUrl}>{group.outline.title}</a>
          </li>
        </ul>
      ) : null}
    </li>
  ))}
  </ul>
